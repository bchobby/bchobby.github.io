1、本“项目”用到的“工具”


  • 模块

    1、random
     用 random 模块的 randint() 方法产生两个随机整数，choice() 方法随机选择四则运算符 “+-*/” 。

    2、time
     用 time() 方法获取系统当前时间秒，以方便计算“练习计算”所用时长和平均解题用时。

  • 内置函数

    eval ，将“四则运算表达式”字符串转换成 Python 对象，得出结果(Python 对象如果是表达式，可以自动计算结果)。

5、完整源码

(源码较长，点此跳过源码)

#!/sdcard/qpython python
# coding: utf-8
import random
#random.seed(0) # 设定随机数种子后，相同序列取值每次都一样。默认以系统时间截为随机种子，每次都不同。
import time


def calculator(n, maxnum):
    ''' 四则运算计算器 '''
    correct = 0 # 答对题目计数器初值。
    tip = f"{'':~^36}" # 分割线字符串。
    print(f"{tip:^50}\n")
    start = time.time() # 保存当前时间秒。
    
    for i in range(n): # 循环练习n道题目。
        a = random.randint(0, maxnum) # 随机生成整数a。
        b = random.randint(1, maxnum) # 随机生成整数b ，b!= 0，避免0为除数。

        if b > a and a != 0:
            a, b = b, a # 如果 b > a ，交换 a、b 的值，a是0时不交换。

        sign = random.choice('+-*/') # 除法结果是浮点数，判定麻烦，这里不用除法。
        exp = f"{a} {sign} {b}" # 四则运算表达式。
        sign = sign if sign not in '*/' else '×' if sign == '*' else '÷' # 如果运算符是'*'，换成'×' ，是/换成÷，+-不变。
        exp_str = f"{a} {sign} {b}" # 表达式打印字符串。
        
        while True: # 输入答案循环。
            try:
                isright = input(f"{title}\n{i+1:>16}. {exp_str} = ").strip()
                isright = float(isright) # 转换键盘输入答案为浮点数类型。
                break # 输入无异常，退出循环。
            except Exception as error:
                input(f"{red}\nErrorType: {error}\n{' 请输入错误！':~^44}{stop}")


        result = eval(exp) # 计算算术表达式结果。

        if isinstance(result, float):
            result = int(result) if not float(str(result).split('.')[1]) else f"{result:.2f}" # 结果小数部分全0，取整；否则，保留两小数。

        if isright == float(result):
            correct += 1 # 累加答对题数。
            tip = f"{' 您真棒！':~^34}"
            input(f"{green}\n{tip:^46}{stop}") # 打印鼓励信息。
        else:
            input(f"{red}\n{'正确答案：':>18}{result}{stop}")

    mytime = time.time() - start
    print(f"{title}\n{green}\n{'':>14}答对 {correct} 题，正确率 {correct/n*100:.2f}% \n{stop}{gray}{'':~^50}\n{'':>8}用时 {mytime:.2f} 秒，平均解题 {mytime/n:.2f} 秒/题\n{stop}")


if __name__ == '__main__':
    clear = '\033[2J' # Linux 清屏字符串。
    red = '\033[31;1m' # Linux 下亮红色控制字符串。
    gray = '\033[2m' # Linux 下亮绿色控制字符串。
    green = '\033[32;1m' # Linux 下亮绿色控制字符串。
    stop = '\033[0m' # Linux 下恢复默认颜色控制字符串。
    title = f"{clear}\n{' 随机出题整数四则运算 ':~^40}\n"

    while True:
        print(title)
        
        try:
            n, maxnum = map(int, [input(f"\n{' ':>8}{i}：") for i in ('练习题目数', '运算最大取值')])
            break # 输入无异常，退出输入循环。
        except Exception as error:
            input(f"{red}\nErrorType: {error}\n{' 请输入正整数！':~^43}{stop}")

    calculator(n, maxnum) # 调用“计算器”函数开始练习。
