  5. 更新数据：所有人工资变成5000
  6. 更新数据：张三薪资变成3000id变成0

代码实现

-- 创建数据库
create database mydb;

使用mydb数据库
use mydb;

-- 创建一个员工表employee 列为 员工编号id 员工姓名name 员工性别gender 员工薪资salary
create table employee(
id int,  -- 员工id
name varchar(20),  -- 员工姓名
gender varchar(5), -- 员工性别
salary int  -- 员工薪资
);

-- 插入数据:张三 男 2000 李四 男 1000 李红 女 4000
insert into employee(name,gender,salary) values
('张三','男',2000),
('李四','男',1000),
('李红','女',4000);

-- 更新数据：所有人工资变成5000
update employee set salary=5000;

-- 更新数据：张三薪资变成3000id变成0
update employee set salary=3000,id=0 where name='张三';

-- 更新数据：所有人工资再加1000
update employee set salary=salary+1000;

-- 删除数据：删除李红的数据
delete from employee where name='李红';

-- 清空表
drop table employee;

代码实现

-- 查询表中所有学生的信息
select * from student;

-- 查询表中所有学生的姓名和英语成绩
select name,english from student;

-- 过滤表中的重复数据
select distinct * from student; 
            -- distinct只能过滤一次

-- 统计每个学生的总分
select name,chinese+english+math as total_score from student; 
            -- 列用sum行用算术运算符即可

-- 所有学生总分上加上10
select name,(chinese+english+math)+10 from student; 
            --由于本身没有总分所以先弄出总分再加10，配用运算符

-- 使用别名表示学生分数
select name '姓名'，chinese '语文'，english '英语' , math '数学' from student;

-- 查询英语成绩大于90分的所有同学
select * from where english>90;

-- 查询总分大于200分的所有同学
select name,math+chinese+english as total_score from student where math+chinese+english>200;
            -- where后面不能直接用别名total_score是因为where优先于select,此时还没有total_score这个别名

-- 查询英语成绩在80到90之间的所有同学
select * from student where english>=80 && english<=90;

-- 查询英语成绩不在80到90之间的所有同学
select * from student where english<80 || english >90;
select * from student where !(english>=80 && english <=90);   -- !和not一样

--  查询数学分数为89，90，91的同学
select * from student where math in(89,90,91);

-- 查询数学分数不为89,90,91的同学
select * from student where math not in(89,90,91);

-- 查询所有姓李的学生英语成绩
select * from student where name like '李%';
            -- manual 手动 corresponds 对应 syntax 语法  报错

-- 查询数学分80且语文80分的同学
select * from student where math=80 && chinese=80;

-- 查询英语80或者总分200的同学
select * from student where english=80 || (math + english + chinese)=200;

-- 对学生数学成绩进行降序排序后输出
select * from student order by math desc;

-- 对总分从高到低的顺序输出
select * from student order by (math+english+chinese) desc;

-- 对姓李的学生按总成绩升序排序输出
select * from student where name like '李%' order by  (math+english+chinese) asc;  
            -- 注意sql的书写顺序

-- 查询男生和女生分别有多少人，并将人数降序输出
select gender,count(*)as rs from student group by gender  order by rs desc ; 

-- 查询男生和女生分别有多少人，并将人数降序输出，并查询出大于9的性别人数信息
select gender,count(*) as rs from student group by gender having rs>9 order by rs desc;

代码实现

-- 按员工编号升序排列不在10号部门工作的员工信息
select * from emp where deptno!=10 order by empno asc;

-- 查询姓名第二个字母不是'A'且薪水大于1000元的员工信息，按薪水降序排列
select * from emp where ename not like  '_A%' && sal>1000 order by sal desc;

-- 按年薪+奖金降序排序
select * from emp where ename not like  '_A%' && sal>1000 order by (sal*12 + ifnull(comm,0))desc;
            -- ifnull的用法 如果comm为null，则当作0来计算，不为null就是原来值计算
            --格式ifnull(,)

-- 求每个部门的平均薪水 
select deptno,avg(sal) from emp group by deptno;
            -- 审题不清,接下来平均薪资降序并降序，接下来改正
select deptno,avg(sal) as avg_sal from emp group by deptno order by avg_sal desc;
            -- 要明白oder by 和 limit 是在select之后运行的,所以select后面的别名可以识别到。

-- 求每个部门的最高薪水
select deptno,max(sal) from emp group by deptno;

-- 求每个部门每个岗位的最高薪水并美观
select deptno,job,max(sal) from emp group by deptno,job order by deptno asc;
            -- 有分组查询出现的时候 select后面只能跟分组字段和聚合函数的字段

-- 求平均薪水大于2000的部门编号
select deptno,avg(sal) from emp group by deptno having avg(sal)>2000;
            -- 又考察到了执行顺序不能用where只能用having 分组之后的字段条件用having

-- 将部门平均薪水大于1500的部门编号列出来，按部门平均薪水降序排列
select deptno,avg(sal) from emp group by deptno having avg(sal)>1500 order by avg(sal) desc;

-- 选择公司中有奖金和员工姓名和工资
select ename,sal+comm as xs from emp where comm is not null;

-- 查询员工最高工资和最低工资的差距
select max(sal),min(sal),max(sal)-min(sal) as cj from emp;
            -- 考察到同一关键字内运行顺序,，接下来改正
select max(sal)-min(sal) as cj from emp;
