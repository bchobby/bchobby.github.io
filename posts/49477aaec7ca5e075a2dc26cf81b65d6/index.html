语句项：

        在switch语句中，我们没办法直接实现分支，搭配break使用才能实现真正的分支。

例如：输入相应数字，对应星期几

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main()
{
	int day = 0;
	scanf("%d", &day);
	switch (day)				//括号内的值必须为整型
	{
	case 1:						//case后面的数为整型常量
		printf("周一\n");
		break;
	case 2:
		printf("周二\n");
		break;
	case 3:
		printf("周三\n");
		break;
	case 4:
		printf("周四\n");
		break;
	case 5:
		printf("周五\n");
		break;
	case 6:
		printf("周六\n");
		break;
	case 7:
		printf("周日\n");
		break;
	default:
			printf("输入错误\n");
		break;
	}
	return 0;
}

输入相应数字，分辨是工作日还是休息日

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main()
{
	int day = 0;
	scanf("%d", &day);
	switch (day)
	{
	case 1:
	case 2:
	case 3:
	case 4:
	case 5:
		printf("工作日\n");
		break;
	case 6:
	case 7:
		printf("休息日\n");
		break;
	default:
		printf("输入错误\n");
	}
	return 0;
}

        代码中加入break后，当运行到break所在位置时，会跳出switch语句，而default语句可以写在任何一个 case 标签可以出现的位置。 当 switch 表达式的值并不匹配所有 case 标签的值时，这个 default 子句后面的语句就会执行。所以，每个switch语句中只能出现一条default子句。 但是它可以出现在语句列表的任何位置，而且语句流会像执行一个case标签一样执行default子句。

        2.2.2 switch语句的嵌套使用：

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main()
{
	int n = 1;
	int m = 2;
	switch (n)
	{
	case 1:
		m++;
	case 2:
		n++;
	case 3:
		switch (n)
		{//switch允许嵌套使用
		case 1:
			n++;
		case 2:
			m++;
			n++;
			break;
		}
	case 4:
		m++;
		break;
	default:
		break;
	}
	printf("m = %d, n = %d\n", m, n);
	return 0;
}

         3.1.3 while语句中的break和continue

        break的作用：停止while循环后期的所有循环，直接永久终止循环

 例：

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main()
{
	int i = 1;
	while (i <= 10)
	{
		if (i == 5)            //当i的值为5时，执行if语句中的break直接跳出while循环，将不再执行 
                                下面的语句
			break;
		printf("%d ", i);
		i = i + 1;
	}
	return 0;
}

 打印结果

        continue的作用:用于终止本次循环的，也就是本次循环中continue后边的代码不会再执行， 而是直接跳转到while语句的判断部分。进行下一次循环的入口判断。

例1：

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main()
{
	int i = 1;
	while (i <= 10)
	{
		if (i == 5)
			continue;            //当i、的值为5时满足if语句，将执行continue语句，跳过本次循环， 
                                   直接进入（i<=10）的判断中，一直到if语句，此时i的值没有改变， 
                                   故又跳过循环，在此判断，这样就陷入死循环
		printf("%d ", i);
		i = i + 1;
	}
	return 0;
}

打印结果：

 例2：

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main()

		if (i == 5)

        3.1.4 用while循环进行不断读取并不断打印(也可以用来清理缓冲区)：

        EOF：文件结束标志，#define下的 -1，接收时用int类型

int main()
{
    int ch = 0;
    while ((ch = getchar()) != EOF)
    {
        putchar(ch); 
    }
    return 0;
}

打印结果：       

       

         自动换行的原因是因为：在输入一个字符后回车，相当于在输入字符后回车又输入了一个 \0 ，getchar是在缓冲区获取数据，而且一次只能获取一个字符，在获取一个字符后，还剩下一个\0，下一次读取就会读取\0，所以会换行。

 例如：

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main()
{
	char password[20] = { 0 };
	printf("请输入密码》\n");
	scanf("%s", password);
	int ch = 0;
	getchar();					//清理输入密码时回车输入的\n
	printf("请确定密码》(Y/N)\n");
	int ret = getchar();
	if ('Y' == ret)
	{
		printf("Yes");
	}
	else
	{
		printf("No");
	}
	return 0;
}

  3.2 for循环

        3.2.1语法

for(表达式1; 表达式2; 表达式3)
 循环语句;

表达式1：初始化部分，用于初始化循环变量的。

表达式2 ：表达式2为条件判断部分，用于判断循环时候终止。

表达式3： 调整部分，用于循环条件的调整。

         for中的break跟while中一样，直接跳出循环，但在while循环中continue是直接跳到判断部分，但在for循环中continue是跳到调整部分。

        3.2.4 while循环与for循环的对比：

while循环
int main()
{
	int i = 0;                  //初始化
	while (i <= 10)             //判断部分
	{
		printf("%d ", i);
		i++;                    //调整部分
	}
	return 0;
}

for循环
int main()
{
	int i = 0;
	//for(i=1（初始化）; i<=10（判断部分）; i++（调整部分）)
	for (i = 1; i <= 10; i++)
	{
		printf("%d ", i);
	}
	return 0;
}

        for循环的优点：可以发现在while循环中依然存在循环的三个必须条件，但是由于风格的问题使得三个部分很可能偏离较远，这样查找修改就不够集中和方便。所以，for循环的风格更胜一筹；for循环使用的频率也最高。将初始化、比较部分、调整部分放到一起 ，能够更清晰的观察能够循环的条件。

        for循环的缺点：由于其初始化、比较、调整部分的高度集中，在通过初始化后进入循环体，完成循环时进入调整部分，紧接着就是比较部分，这是一个固定的路线，不像while循环那么灵活。而且在for循环的循环体中，不可在for 循环体内修改循环变量，防止 for 循环失去控制。

        3.3.2 执行流程

        3.3.3  do......while语句中的break、continue

        在该语句中，执行break会永久跳出循环，执行continue时会直接跳到判断部分。

break:

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main()
{
	int i = 10;

	do
	{
        i++;
		if (7 == i)
			break;
		printf("%d\n", i);
	} while (i < 10);

	return 0;
}

打印结果：

        当执行到 if（7==i）时，执行break语句，跳出循环。

 continue:

#define _CRT_SECURE_NO_WARNINGS

int main()
{
	int i = 1;

	do
	{
		i++;
		if (7 == i)
			continue;

四、拓展：

        goto语句：C语言中提供了可以随意滥用的 goto语句和标记跳转的标号。 从理论上 goto语句是没有必要的，实践中没有goto语句也可以很容易的写出代码。 但是某些场合下goto语句还是用得着的，最常见的用法就是终止程序在某些深度嵌套的结构的处理过程。

例如：一次跳出两层或多层循环。
多层循环这种情况使用 break 是达不到目的的。它只能从最内层循环退出到上一层的循环。
