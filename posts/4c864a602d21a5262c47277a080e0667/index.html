Java JDBC连接各种数据库的配置方法

 

MySQL：

Java代码  
  1. String Driver="com.mysql.jdbc.Driver";    //驱动程序  
  2. //连接的URL,db_name为数据库名  
  3. String URL="jdbc:mysql://localhost:3306/db_name";       
  4. String Username="username";    //用户名  
  5. String Password="password";    //密码  
  6. Class.forName(Driver);  
  7. Connection con=DriverManager.getConnection(URL,Username,Password);  

 Microsoft SQL Server 2.0驱动(3个jar):

 

Java代码  
  1. //连接SQL数据库的方法  
  2. String Driver="com.microsoft.jdbc.sqlserver.SQLServerDriver";       
  3. //db_name为数据库名  
  4. String URL="jdbc:microsoft:sqlserver://localhost:1433;DatabaseName=db_name";       
  5. String Username="username";    //用户名  
  6. String Password="password";    //密码  
  7. Class.forName(Driver);    //加载数据可驱动  
  8. Connection con=DriverManager.getConnection(URL,UserName,Password);  

 Microsoft SQL Server 3.0驱动(1个jar): 

 

Java代码  
  1. //连接SQL数据库的方法  
  2. String Driver="com.microsoft.sqlserver.jdbc.SQLServerDriver";       
  3. //db_name为数据库名  
  4. String URL="jdbc:microsoft:sqlserver://localhost:1433;DatabaseName=db_name";       
  5. String Username="username";    //用户名  
  6. String Password="password";    //密码  
  7. Class.forName(Driver);    //加载数据可驱动  
  8. Connection con=DriverManager.getConnection(URL,UserName,Password);  

 Oracle(用thin模式):

 

Java代码  
  1. String Driver="oracle.jdbc.driver.OracleDriver";    //连接数据库的方法  
  2. String URL="jdbc:oracle:thin:@loaclhost:1521:orcl";    //orcl为数据库的SID  
  3. String Username="username";    //用户名  
  4. String Password="password";    //密码  
  5. Class.forName(Driver);    //加载数据库驱动  
  6. Connection con=DriverManager.getConnection(URL,Username,Password);  

 DB2：

 

Java代码  
  1. //连接具有DB2客户端的Provider实例  
  2. String Driver="com.ibm.db2.jdbc.app.DB2.Driver";      
  3. //db_name为数据可名  
  4. String URL="jdbc:db2://localhost:5000/db_name";       
  5. String Username="username";    //用户名  
  6. String Password="password";    //密码  
  7. Class.forName(Driver);  
  8. Connection con=DriverManager.getConnection(URL,Username,Password);  

 PostgreSQL:

 

Java代码

  2. //db_name为数据可名  
  3. String URL="jdbc:postgresql://localhost/db_name";       
  4. String Username="username";    //用户名  
  5. String Password="password";    //密码  
  6. Class.forName(Driver);

 JDBC-ODBC:

 

Java代码  
  1. String Driver="sun.jdbc.odbc.JdbcOdbcDriver";  
  2. String URL="jdbc:odbc:dbsource";    //dbsource为数据源名  
  3. String Username="username";    //用户名  
  4. String Password="password";    //密码  
  5. Class.forName(Driver);  
  6. Connection con=DriverManager.getConnection(URL,Username,Password);
