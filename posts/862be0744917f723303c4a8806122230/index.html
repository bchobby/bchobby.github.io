% Compute the T2 statistic, SPE statistic,and the nonlinear component of Y
%
%Â  Â Â  Â  [SPE,T2,mappedY] = kpca_test(model,Y)
%
% INPUT
%Â  Â modelÂ  Â Â  Â  KPCA model

%
% OUTPUT
