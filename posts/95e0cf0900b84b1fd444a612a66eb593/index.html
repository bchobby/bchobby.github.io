# 多重定位
ele_table=driver.find_element_by_id(u'table')

may=[]
june=[]
for i in range(len(ele_table_trs)):
    if i==0:
        continue
    else:
        tds=ele_table_trs[i].find_elements_by_tag_name(u'td')
        for j in range(len(tds)):
            if j==1:
                may.append(tds[j].text)
            elif j==2:
                june.append(tds[j].text)
            else:
                continue
print may
print june
may_sum=0
june_sum=0
for m in may:
    may_sum+=int(m)
for j in june:
    june_sum+=int(j)
