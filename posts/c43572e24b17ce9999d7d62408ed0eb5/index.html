创建容器

# zookeeper
docker run -it -d --net fixed --ip 172.18.0.2 \
-p 2182:2181 --name zoo zookeeper

# dubbo-admin
docker run -it -d --net fixed --ip 172.18.0.3 -p 8000:8080 \
-e admin.registry.address=zookeeper://172.18.0.2:2181 \
-e admin.config-center=zookeeper://172.18.0.2:2181 \
-e admin.metadata-report.address=zookeeper://172.18.0.2:2181 \
--name dubbo-admin apache/dubbo-admin
