创建矩阵、查看维度、形状和元素数量

import numpy as np

array = [[1,2,3],
         [4,5,6]]
array = np.array(array)
print(array)
print('number of dim: ', array.ndim)
print('shape: ', array.shape)
print('size: ', array.size)

 [4 5 6]]

shape:  (2, 3)
size:  6
