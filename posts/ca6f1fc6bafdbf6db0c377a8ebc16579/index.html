 *
 * @param  {number} val Non-negative value.

function nice(val, round) {
  var exponent = Math.floor(Math.log(val) / Math.LN10);
  var exp10 = Math.pow(10, exponent);
  var f = val / exp10; // 1 <= f < 10

  var nf;

  if (round) {

      nf = 1;
    } else if (f < 2.5) {
      nf = 2;
