db2数据库联邦，WRAPPER，SERVER，MAPPING，NICKNAME，10.5

注：名称长度不超过8位

 

##切换用户启动数据库

su - db2inst1

db2start

 

##连接数据库

db2 connect to 本地数据库名称 user 用户名 using 密码;

db2 set schema 本地SCHEMA;

 

 

## 打开联邦数据库参数

db2 update dbm cfg using FEDERATED YES

db2 force application all

db2stop

db2start

 

## 编目远程数据库节点

db2 "catalog tcpip node 节点名称 remote IP server 端口"

db2 "terminate"

 

## 编目远程数据库

db2 "catalog db 远程数据库名称 as 远程数据库在本地别名  at node 节点名称 authentication SERVER"

db2 "terminate"

 

## 测试连接是否可用

db2 "list db directory"

db2 "connect to 远程数据库在本地别名 user 用户名 using 密码"

 

## 创建数据库联邦

db2 connect to 本地数据库名称 user 用户名 using 密码;

 

## 创建WRAPPER （包容器）

db2 "CREATE WRAPPER 包容器名称"

db2 "CREATE WRAPPER DRDA"

 

 

## 创建连接对方数据的SERVER

db2 "CREATE SERVER SERVER名称 TYPE DB2/UDB VERSION '10.5' WRAPPER 包容器名称 AUTHORIZATION \"用户名\" PASSWORD \"密码\" OPTIONS (ADD NODE '节点名称',DBNAME '远程数据库在本地别名')"

 

## 创建MAPPING（创建用户映射，注意是用户映射）

db2 "CREATE USER MAPPING FOR 本地用户名 SERVER SERVER名称  OPTIONS (ADD REMOTE_AUTHID '用户名',REMOTE_PASSWORD '密码')"

 

## 创建联邦关系表

db2 "CREATE NICKNAME 本地SCHEMA.远程表别名 FOR SERVER名称.远程SCHEMA.远程表名"

 

#查询

db2 "select * from 本地SCHEMA.远程表别名"
